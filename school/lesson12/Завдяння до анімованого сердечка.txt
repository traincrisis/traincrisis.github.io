HTML
1. Створити блок (class = "photo") і вставити на нього фото love.png
2. Створити блок (class = "text") і розмістити на ньому текст 
"With love for you..."

CSS

.photo{
	ширина 300 пікселів
	зовнішні відступи 30 пікселів
}
фото{
	ширина 100%		
}
фото:при_наведенні{
	назва анімації
	тривалість 3 секунди
	безперервне повторення
}
.text{
	фоновий колір rgb(247, 105, 105)
	ширина 300 пікселів
	внутрішні та зовнішні відступи 30 пікселів
	анімація: ім'я тривалість(5s) постійно_повторюється повертається_назад	
}
h1{
	сімейство шрифтів курсив(cursive)
	стиль букв курсивний
	анімація ім'я 5секунд безперервна повертається_назад
}

@ключові_кадри назва_анімації_для_сердечка{
	0%{
		розмір як є (transform: scale(1);)
	}
	100%{
		збільшуємо на 0.3 ( transform: scale(1.3);)
	}
} 
анімація_для_тексту{
	from{
		розмір тексту 50%
	}
	40%{
		колір lavender
	}
	to{
		розмір тексту 180%
	}
}
анімація для блоку з текстом{
	0%{
		прозорість 0.1
		фоновий колір antiquewhite 
		заокруглення кутів 3 пікселі
	}
	40%{
		фоновий колір чорний
	}
	70%{
		фоновий колір світлорожевий
	}
	100%{
		фоновий колір насичений червоний
		прозорість 1
		радіус заокруглення 100 пікселів
	}
}
